# 📄 FICHIER 1: bot.js

```javascript
import { Client, GatewayIntentBits, REST, Routes, SlashCommandBuilder, EmbedBuilder } from 'discord.js';
import dotenv from 'dotenv';

dotenv.config();

const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages,
    ]
});

const TOKEN = MTUwNDc3NjM3MTc0ODk5NTEwMQ.GneK0U.dGJHj2kwjsillwJr8QZyqEKp_0McE2ADaBA8XE;
const CLIENT_ID = 1504776371748995101;

// Fonction pour générer un code Nitro aléatoire (format standard Discord)
function generateNitroCode() {
    const characters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
    let code = '';
    for (let i = 0; i < 16; i++) {
        code += characters.charAt(Math.floor(Math.random() * characters.length));
    }
    return code;
}

// Fonction pour vérifier un code Nitro (simulation - l'API Discord ne permet pas vraiment de tester)
async function checkNitroCode(code) {
    try {
        // AVERTISSEMENT: Discord n'a pas d'API publique pour vérifier les codes Nitro
        // Cette tentative utilisera l'endpoint de promotion qui est techniquement non-documenté
        // et pourrait entraîner un bannissement
        
        const response = await fetch(`https://discord.com/api/v10/entitlements/gift-codes/${code}`, {
            method: 'GET',
            headers: {
                'Content-Type': 'application/json'
            }
        });

        if (response.status === 200) {
            const data = await response.json();
            return { valid: true, data };
        } else if (response.status === 404) {
            return { valid: false };
        } else if (response.status === 429) {
            // Rate limited
            return { valid: false, rateLimited: true };
        } else {
            return { valid: false };
        }
    } catch (error) {
        console.error('Erreur lors de la vérification:', error);
        return { valid: false, error: true };
    }
}

// Fonction principale pour chercher un code valide
async function findValidNitroCode(interaction) {
    let attempts = 0;
    const maxAttempts = 100; // Limiter pour éviter les abus
    let found = false;
    let validCode = null;

    const embed = new EmbedBuilder()
        .setColor('#5865F2')
        .setTitle('🔍 Recherche de code Nitro en cours...')
        .setDescription('Cette opération peut prendre du temps.')
        .setFooter({ text: '⚠️ Attention: Cette pratique viole les ToS de Discord' })
        .setTimestamp();

    await interaction.editReply({ embeds: [embed] });

    while (attempts < maxAttempts && !found) {
        attempts++;
        const code = generateNitroCode();
        
        // Mise à jour tous les 10 essais
        if (attempts % 10 === 0) {
            const progressEmbed = new EmbedBuilder()
                .setColor('#FFA500')
                .setTitle('🔍 Recherche en cours...')
                .setDescription(`**Tentatives:** ${attempts}/${maxAttempts}\n**Dernier code testé:** ||${code}||`)
                .setFooter({ text: '⚠️ Attention: Cette pratique viole les ToS de Discord' })
                .setTimestamp();
            
            await interaction.editReply({ embeds: [progressEmbed] });
        }

        const result = await checkNitroCode(code);

        if (result.rateLimited) {
            const rateLimitEmbed = new EmbedBuilder()
                .setColor('#FF0000')
                .setTitle('❌ Rate Limit Atteint')
                .setDescription('Discord a bloqué temporairement les requêtes. Réessayez plus tard.')
                .setFooter({ text: '⚠️ Ceci est exactement pourquoi cette pratique est interdite' })
                .setTimestamp();
            
            await interaction.editReply({ embeds: [rateLimitEmbed] });
            return;
        }

        if (result.valid) {
            found = true;
            validCode = code;
        }

        // Petit délai pour éviter de spam l'API Discord
        await new Promise(resolve => setTimeout(resolve, 500));
    }

    if (found) {
        const successEmbed = new EmbedBuilder()
            .setColor('#00FF00')
            .setTitle('✅ Code Nitro Trouvé!')
            .setDescription(`**Code valide:** ||${validCode}||\n**Lien:** https://discord.com/gifts/${validCode}`)
            .addFields(
                { name: 'Tentatives', value: attempts.toString(), inline: true },
                { name: 'Statut', value: '✅ Valide', inline: true }
            )
            .setFooter({ text: '⚠️ Utilisez ce code rapidement avant qu\'il ne soit réclamé' })
            .setTimestamp();
        
        await interaction.editReply({ embeds: [successEmbed] });
    } else {
        const failEmbed = new EmbedBuilder()
            .setColor('#FF0000')
            .setTitle('❌ Aucun Code Trouvé')
            .setDescription(`Aucun code valide trouvé après ${attempts} tentatives.`)
            .addFields(
                { name: 'Tentatives', value: attempts.toString(), inline: true },
                { name: 'Statut', value: '❌ Échec', inline: true }
            )
            .setFooter({ text: 'Les codes Nitro valides sont extrêmement rares' })
            .setTimestamp();
        
        await interaction.editReply({ embeds: [failEmbed] });
    }
}

// Commande Slash
const commands = [
    new SlashCommandBuilder()
        .setName('nitro')
        .setDescription('⚠️ Tente de trouver un code Nitro valide (VIOLE LES ToS DE DISCORD)')
        .toJSON()
];

// Enregistrer les commandes
const rest = new REST({ version: '10' }).setToken(TOKEN);

(async () => {
    try {
        console.log('Enregistrement des commandes slash...');
        await rest.put(
            Routes.applicationCommands(CLIENT_ID),
            { body: commands }
        );
        console.log('✅ Commandes enregistrées avec succès!');
    } catch (error) {
        console.error('Erreur lors de l\'enregistrement des commandes:', error);
    }
})();

// Event: Bot prêt
client.once('ready', () => {
    console.log(`✅ Bot connecté en tant que ${client.user.tag}`);
    console.log('⚠️  AVERTISSEMENT: Ce bot viole les conditions d\'utilisation de Discord!');
    console.log('⚠️  L\'utilisation de ce bot peut entraîner un bannissement permanent.');
});

// Event: Interaction (commande slash)
client.on('interactionCreate', async (interaction) => {
    if (!interaction.isChatInputCommand()) return;

    if (interaction.commandName === 'nitro') {
        // Avertissement initial
        const warningEmbed = new EmbedBuilder()
            .setColor('#FF0000')
            .setTitle('⚠️ AVERTISSEMENT IMPORTANT')
            .setDescription(
                '**Cette commande viole les conditions d\'utilisation de Discord!**\n\n' +
                '🚫 Risques:\n' +
                '• Bannissement permanent de votre compte\n' +
                '• Bannissement de votre IP\n' +
                '• Actions légales possibles\n\n' +
                '**Voulez-vous vraiment continuer?**'
            )
            .setFooter({ text: 'Utilisez cette commande à vos propres risques' })
            .setTimestamp();

        await interaction.reply({ embeds: [warningEmbed], ephemeral: true });

        // Attendre 3 secondes puis démarrer
        setTimeout(async () => {
            await findValidNitroCode(interaction);
        }, 3000);
    }
});

// Démarrer le bot
client.login(TOKEN);
```

# 📄 FICHIER 7: start-bot.sh (Mac/Linux)

```bash
#!/bin/bash

echo "========================================"

echo "Node.js détecté: $(node --version)"

# Vérifier les dépendances

if [ ! -d "node_modules" ]; then
    echo "Installation des dépendances..."
    npm install
fi
echo ""

# Vérifier le fichier .env
echo "Vérification du fichier .env..."
if [ ! -f ".env" ]; then
    echo "ERREUR: Fichier .env introuvable!"

    echo ""
    read -p "Voulez-vous créer le fichier maintenant? (o/n) " -n 1 -r
    echo ""
    if [[ $REPLY =~ ^[Oo]$ ]]; then
        cp .env.example .env
        echo "Fichier .env créé! Modifiez-le avec vos informations."
        ${EDITOR:-nano} .env
        echo ""
        read -p "Appuyez sur Entrée une fois le fichier configuré..."

        exit 1
    fi
fi
echo "Fichier .env trouvé!"
echo ""

echo "========================================"
echo "Démarrage du bot..."
echo "========================================"
echo ""

node bot.js

if [ $? -ne 0 ]; then
    echo ""
    echo "ERREUR: Le bot s'est arrêté avec une erreur."
    echo "Vérifiez votre configuration dans .env"
    read -p "Appuyez sur Entrée pour quitter..."
fi
```

**Instructions:**

# 🚀 GUIDE D'INSTALLATION COMPLET

## Étape 1: Préparer l'environnement

### Installer Node.js
1. Allez sur https://nodejs.org/
2. Téléchargez la version LTS (recommandée)
3. Installez Node.js
4. Vérifiez l'installation:
   ```bash
   node --version
   npm --version
   ```

---

## Étape 2: Créer l'Application Discord

### 1. Créer l'Application
1. Allez sur https://discord.com/developers/applications
2. Cliquez sur **"New Application"**
3. Nommez votre bot (ex: "NitroBot")
4. Cliquez sur **"Create"**

### 2. Créer le Bot
1. Menu de gauche → **"Bot"**
2. Cliquez sur **"Add Bot"**
3. Confirmez **"Yes, do it!"**

### 3. Récupérer le Token
1. Section TOKEN
2. Cliquez sur **"Reset Token"**
3. Cliquez sur **"Copy"**
4. **⚠️ GARDEZ-LE SECRET!**

### 4. Activer les Intents
Descendez jusqu'à "Privileged Gateway Intents" et activez:
- ☑️ **Presence Intent**
- ☑️ **Server Members Intent**
- ☑️ **Message Content Intent**
- Cliquez sur **"Save Changes"**

### 5. Récupérer le Client ID
1. Menu de gauche → **"OAuth2"** → **"General"**
2. Copiez **"APPLICATION ID"**

---

## Étape 3: Inviter le Bot

### Générer l'URL d'Invitation
1. Menu de gauche → **"OAuth2"** → **"URL Generator"**
2. **SCOPES:** Cochez
   - ☑️ `bot`
   - ☑️ `applications.commands`
3. **BOT PERMISSIONS:** Cochez
   - ☑️ Send Messages
   - ☑️ Use Slash Commands
   - ☑️ Embed Links
4. Copiez l'URL générée en bas
5. Ouvrez-la dans votre navigateur
6. Sélectionnez votre serveur
7. Cliquez sur **"Authorize"**

# 🔧 DÉPANNAGE

## ❌ Erreur: "Invalid Token"

**Causes:**
- Token incorrect dans `.env`
- Espaces avant/après le token
- Guillemets autour du token

**Solution:**
1. Vérifiez `.env`
2. Régénérez le token sur le Developer Portal
3. Pas d'espaces, pas de guillemets

---

## ❌ Le bot ne répond pas

**Solutions:**
1. **Attendez 5-10 minutes** (synchronisation des commandes)
2. Vérifiez que le bot est en ligne (point vert)
3. Vérifiez les permissions du bot
4. Redémarrez le bot (Ctrl+C puis `node bot.js`)

---

## ❌ "Cannot find module discord.js"

**Solution:**
```bash
npm install
```

Si ça ne fonctionne pas:
```bash
npm install discord.js dotenv
```

---

## ❌ Le bot se déconnecte immédiatement

**Cause:** Intents non activés

**Solution:**
1. Developer Portal → Bot
2. Activez les 3 "Privileged Gateway Intents"
3. Sauvegardez
4. Redémarrez le bot

---

## ❌ "Rate Limited"

**C'est normal!** Discord limite les requêtes.

**Solution:** Attendez 10-15 minutes
